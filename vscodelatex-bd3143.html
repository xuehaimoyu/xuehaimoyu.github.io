VScode编译latex记录

Posted on Wed, May 1, 2024 latex record

本文主要记录使用vscode编译latex的方法，希望能有所帮助。

毕业设计参考模版：

感谢前辈大佬们的付出。

具体步骤：

（1）下载 Texlive。

时间可能比较久。

（2）下载VScode。

（3）下载VScode的latex插件：

latex workshop

然后打开设置：

粘贴：

{
    //------------------------------LaTeX 配置----------------------------------
       // 设置是否自动编译
       "latex-workshop.latex.autoBuild.run":"never",
       //右键菜单
       "latex-workshop.showContextMenu":true,
       //从使用的包中自动补全命令和环境
       "latex-workshop.intellisense.package.enabled": true,
       //编译出错时设置是否弹出气泡设置
       "latex-workshop.message.error.show": false,
       "latex-workshop.message.warning.show": false,
       // 编译工具和命令
       "latex-workshop.latex.tools": [
           {
               "name": "xelatex",
               "command": "xelatex",
               "args": [
                   "-synctex=1",
                   "-interaction=nonstopmode",
                   "-file-line-error",
                   "-shell-escape",//新加入的选项
                   "%DOCFILE%"
               ]
           },
           {
               "name": "pdflatex",
               "command": "pdflatex",
               "args": [
                   "-synctex=1",
                   "-interaction=nonstopmode",
                   "-file-line-error",
                   "-shell-escape",//新加入的选项
                   "%DOCFILE%"
               ]
           },
           

           {
            "name": "latexmk",
            "command": "latexmk",
            "args": [
              "-synctex=1",
              "-interaction=nonstopmode",
              "-file-line-error",
              "-shell-escape",//新加入的选项
              "-pdf",
              "-outdir=%OUTDIR%",
              "%DOC%"
            ],
            "env": {}
          },

          {
            "name": "lualatex",
            "command": "lualatex",
            "args": [
                "-synctex=1",
                "-interaction=nonstopmode",
                
                "-file-line-error",
                "-shell-escape",//这个命令行在网上的Latex Workshop设置里一般没有，所以直接recipe会报错
                "%DOCFILE%"
            ]
            },
            {
                "name": "xelatexmk",
                "command": "latexmk",
                "args": [
                    "-xelatex",
                    "-synctex=1",
                    "-quiet",
                    "-interaction=nonstopmode",
                    "-file-line-error",
                    "-halt-on-error",
                    "-shell-escape",
                    "%DOC%"
                ]
            },

           {
               "name": "bibtex",
               "command": "bibtex",
               "args": [
                   "%DOCFILE%"
               ]
           }
       ],
       // 用于配置编译链
       "latex-workshop.latex.recipes": [
        {
            "name": "PDFLaTeX",
            "tools": [
                "pdflatex"
            ]
        },
        {
            "name": "LuaLaTeX",
            "tools": [
                "lualatex"
            ]
        },
           {
               "name": "XeLaTeX",
               "tools": [
                   "xelatex"
               ]
           },
           {
               "name": "BibTeX",
               "tools": [
                   "bibtex"
               ]
           },
           {
               "name": "LaTeXmk",
               "tools": [
                   "latexmk"
               ]
           },
           {
               "name": "xelatex -> bibtex -> xelatex*2",
               "tools": [
                   "xelatex",
                   "bibtex",
                   "xelatex",
                   "xelatex"
               ]
           },
           {
               "name": "pdflatex -> bibtex -> pdflatex*2",
               "tools": [
                   "pdflatex",
                   "bibtex",
                   "pdflatex",
                   "pdflatex"
               ]
           },
           {
            "name": "latexmk (xelatex)",
            "tools": [
                "xelatexmk"
            ]
        }
       ],
       //文件清理。此属性必须是字符串数组
       "latex-workshop.latex.clean.fileTypes": [
           "*.aux",
           "*.bbl",
           "*.blg",
           "*.idx",
           "*.ind",
           "*.lof",
           "*.lot",
           "*.out",
           "*.toc",
           "*.acn",
           "*.acr",
           "*.alg",
           "*.glg",
           "*.glo",
           "*.gls",
           "*.ist",
           "*.fls",
           "*.log",
           "*.fdb_latexmk"
       ],
       //设置为onFaild 在构建失败后清除辅助文件
       "latex-workshop.latex.autoClean.run": "onFailed",
       // 使用上次的recipe编译组合
    "latex-workshop.latex.recipe.default": "lastUsed",
       //使用 SumatraPDF 预览编译好的PDF文件
       // 设置VScode内部查看生成的pdf文件
       "latex-workshop.view.pdf.viewer": "tab",
       // PDF查看器用于在\ref上的[View on PDF]链接
    "latex-workshop.view.pdf.ref.viewer": "auto",
    "editor.wordWrap": "on",
    "editor.renderControlCharacters": false,
    "Codegeex.Comment.LanguagePreference": "中文",
    "Codegeex.Explanation.LanguagePreference": "中文",
    "Codegeex.SidebarUI.LanguagePreference": "中文",
    "workbench.colorTheme": "Quiet Light",
    "Codegeex.Privacy": false,
    "editor.codeActionsOnSave": {

    }
 }

一些小Tips：

注意如果参考文献编译不通过的话，将模版中的tongjithesis的设置参考文献格式：backend=biber改为backend=bibtex。

编译的时候选xelatex-bibtex-xelatex。
